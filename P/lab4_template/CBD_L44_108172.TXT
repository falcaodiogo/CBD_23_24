# 9. Liste os artistas cujo a música é escrita em mode Major, e cujo as streams na apple são superiores a 200, desde que sejam superiores ao deezer.
MATCH(a:Artist)-[h:HAS_STREAMS]-(s:StreamsAndPlaylists)-[h1:HAS_STREAMS]-(m:Music)
WHERE m.mode = "Major" and s.deezerplaylist < s.apple and s.apple > 200
RETURN DISTINCT m.name

# 10. Mostre as músicas que foram lançandas em fevereiro, independetemente do ano e dia e que tenham mais do que 795686 streams.

WHERE m.month = "2" AND s.streams > "795686"
RETURN DISTINCT m.name
